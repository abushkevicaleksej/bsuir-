//              нечёткую позитивную дугу

#include "scp_keynodes.scsy"

/////////////////////////////////////////////////////
// текст программы
program(gen_sample,
        [[ S = {}       // здесь объявляем исходное множество
        ]],
        [{ e,           // переменная для элемента непределённого типа
           node,        // переменная для узла
           fuzzy_arc,   // переменная для константной нечёткой дуги
           positive_arc // переменная для константной позитивной дуги
        }])

/////////////////////////////////////////////////////
// с помощью оператора genEl создаётся константный
// элемент неопределённого типа в качестве значения
// переменной e
genEl([1_: assign_: const_: undf_: e])

// атрибут assign_ говорит о том, что значение операнда,
// являющегося переменной должно быть определено
// в процессе реализации оператора, то есть в данном
// случае - сгенерировано

/////////////////////////////////////////////////////
// с помощью оператора genElStr5 создаются:
// константный узел, выходящая из него константная
// нечёткая дуга и входящая в неё из узла S
// константная позитиная дуга, являющиеся
// значениями переменных node, fuzzy_arc и
// positive_arc соответственно
genElStr5([1_: assign_: const_: node,
           2_: assign_: const_: fuz_: fuzzy_arc,
           3_:  fixed_: e,
           4_: assign_: const_: pos_: positive_arc,
           5_:  fixed_: S])

/////////////////////////////////////////////////////
// с помощью оператора genElStr3 создаётся
// константная позитиная дуга, являющаяся
// значением positive_arc
genElStr3([1_:  fixed_: S,
           2_: assign_: const_: pos_: positive_arc,
           3_:  fixed_: node])

/////////////////////////////////////////////////////
// с помощью оператора printEl выводим информацию
// о типе значений операндов S и fuzzy_arc, а также о
// типе элементов, входящих в их видимую окрестность
// по входящим и выходящим дугам
printEl([1_: fixed_: S])
printEl([1_: fixed_: fuzzy_arc])

// оператор printEl используется наряду с операторами
// printNl и print в целях отладки

return()
end
