/////////////////////////////////////////////////////
// текст программы
program(erase_sample,
        [[ S, // объявляем множество S
           C  // объявляем множество C
        ]],
        [{ elmn,
           positive_arc_to_arc,
           positive_arc_to_elmn
        }])

/////////////////////////////////////////////////////
// с помощью оператора searchElStr5 ищется элемент,
// входящий в S под атрибутом S
// становится значением переменной result
searchElStr5([1_:  fixed_: S,
              2_: assign_: const_: pos_: positive_arc_to_elmn,
              3_: assign_: elmn,
              4_: assign_: const_: pos_: positive_arc_to_arc,
              5_:  fixed_: C])

/////////////////////////////////////////////////////
// с помощью оператора eraseEl удаляется найденный
// элемент
eraseEl([1_: f_: fixed_: elmn])

// предыдущие два оператора можно заменить на один
// оператор:
// eraseElStr5([1_:  fixed_: S,
//              2_: assign_: f_: const_: pos_: positive_arc_to_elmn,
//              3_: assign_: f_: elmn,
//              4_: assign_: f_: const_: pos_: positive_arc_to_arc,
//              5_:  fixed_: C])

/////////////////////////////////////////////////////
// с помощью оператора eraseElStr3 удаляется элемент,  
// включённый в C
eraseElStr3([1_:  fixed_: C,
             2_: assign_: const_: pos_: f_: positive_arc_to_elmn,
             3_: assign_: f_: elmn])

/////////////////////////////////////////////////////
// с помощью оператора eraseSetStr3 удаляются все  
// дуги, включённые в S
eraseSetStr3([1_:  fixed_: S,
              2_: assign_: const_: pos_: f_: positive_arc_to_elmn,
              3_: assign_: arc_: f_: elmn])

printEl([1_: fixed_: S])
 
return()
end
