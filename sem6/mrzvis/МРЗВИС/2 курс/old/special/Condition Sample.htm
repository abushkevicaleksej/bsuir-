// текст программы
program(condition_sample,
        [[ S, // объявляем множество S
           C  // объявляем множество C
        ]],
        [{ var
        }])

label(cs_first_condition)
ifVarAssign([1_: var],cs_second_condition)

varAssign([1_: assign_: var,
           2_:  fixed_: S],cs_first_condition,cs_first_condition)

label(cs_second_condition)
ifCoin([1_: fixed_: var,
        2_: fixed_: S],,cs_erase_var_value)

varAssign([1_: assign_: var,
           2_:  fixed_: C],cs_first_condition,cs_first_condition)

label(cs_erase_var_value)
varErase([1_: fixed_: var])

return()
end
