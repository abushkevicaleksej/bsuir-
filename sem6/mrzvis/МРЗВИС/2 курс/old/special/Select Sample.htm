#include "data.scs"    

/////////////////////////////////////////////////////
// текст программы
program(select_sample,
        [[ S, // объявляем множество S
           C  // объявляем множество C
        ]],
        [{ elmn,
           node,
           attr_node,
           positive_arc_to_arc,
           positive_arc_to_elmn
        }])

/////////////////////////////////////////////////////
// с помощью оператора selectYStr5 во множестве S
// оставляем только те узлы, которые вкдючают
// хотя бы одну константную позитивную дугу
selectYStr5([1_: assign_: node,
             2_: assign_: const_: pos_: positive_arc_to_elmn,
             3_: assign_: elmn,
             4_: assign_: const_: pos_: positive_arc_to_arc,
             5_: assign_: attr_node,
             set5_:  fixed_: S])

/////////////////////////////////////////////////////
// с помощью оператора printEl выводим информацию
// о типе значения операнда S, а также о
// элементов, входящих в его видимую окрестность
// по входящим и выходящим дугам
printEl([1_: fixed_: S])
 
/////////////////////////////////////////////////////
// с помощью оператора selectNStr3 во множестве S
// оставляем только те элементы, которые не входят
// во множество C
selectNStr3([1_:  fixed_: C,
             2_: assign_: const_: pos_: positive_arc_to_elmn,
             3_: assign_: elmn,
             set3_: fixed_: S])

/////////////////////////////////////////////////////
// с помощью оператора printEl выводим информацию
// о типе значения операнда S, а также о типе
// элементов, входящих в его видимую окрестность
// по входящим и выходящим дугам
printEl([1_: fixed_: S])
 
return()
end
