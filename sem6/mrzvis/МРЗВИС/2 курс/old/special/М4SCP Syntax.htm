Синтаксис содержимого файлов формата m4scp.

<содержимое файла m4scp> ::=

[<пары синонимов>] [<включения>] (<программа на m4scp> |
 <процедура на m4scp>) {<процедура на m4scp>}

<пары синонимов> ::=

<пара синонимов>{ <пара синонимов>}

<пара синонимов> ::=

<идентификатор> = <идентификатор>;

<включения> ::=

<включение> {<включение>}

<включение> ::=

#include "<имя файла>"

<программа на m4scp> ::=

<заголовок программы> [<операторы>] <окончание>

<процедура на m4scp> ::=

<заголовок процедуры> [<операторы>] <окончание>

<заголовок программы> ::=

program(<имя программы>[, [[<множество констант>]][, [<множество переменных>]]])

<заголовок процедуры> ::=

procedure(<имя процедуры>[, [[<множество констант>]][, [[<множество переменных>]][, [<формальные параметры>]]]])

<имя программы> ::=

<идентификатор>

<имя процедуры> ::=

<идентификатор>

<множество констант> ::=

<система множеств>

<множество переменных> ::=

<множество>

<формальные параметры> ::=

<связка параметров>

<система множеств> ::=

[{<идентификаторы>[;]}]

<множество> ::=

{<идентификаторы>}

<идентификаторы> ::=

<идентификатор> [ , <идентификатор > ]

<связка параметров> ::=

{<числовой атрибут>: [in_:] [out_:] <идентификатор>
{, <числовой атрибут >: [in_:] [out_:] <идентификатор> } }

<числовой атрибут> ::=

<натуральное число>_

<натуральное число> ::=

(1|2|3|4|5|6|7|8|9){<цифра>}

<цифра> ::=

1|2|3|4|5|6|7|8|9|0

<окончание> ::=

[<метка>] return[()] end[()]

<метка> ::=

label(<идентификатор>)

<ловушка> ::=

catch(<идентификатор>)

<операторы> ::=

<оператор>{<оператор>}

<оператор> ::=

[<метка>] [<ловушка>]
(<оператор с одним операндом>|
<оператор с двумя операндами>|
<оператор callReturn>|
<оператор с тремя операндами>|
<оператор вида ElStr5>|
<оператор вида SetStr3>|
<оператор вида SetStr5>)

<оператор с одним операндом> ::=

(sys_get_autosegment|
sys_get_default_segment|
sys_set_default_segment|
sys_open_dir|
sys_open_segment|
sys_close_segment|
sys_unlink|
genEl|
print|
printNl|
printEl|
eraseEl|
ifType|
varErase|
contErase|
waitReturn|
ifVarAssign|
ifFormCont)
([[segc_1_: <операнд оператора>,]
1_: <операнд оператора>][<переходы>])

<оператор с двумя операндами> ::=

(sys_get_location|
sys_spin_segment|
sys_wait|
varAssign|
ifCoin|
ifEq|
ifGr|
ifGrEq|
contAssign|
idtfAssign|
idtfMove)
([[segc_1_: <операнд оператора>,]
1_: <операнд оператора>,
[segc_2_: <операнд оператора>,]
2_: <операнд оператора>][<переходы>])

<оператор callReturn> ::=

callReturn([1_: <операнд оператора>,
2_: <операнд оператора>], <идентификатор>)

<оператор с тремя операндами> ::=

(sys_set_event_handler|
call|
add|
sub|
div|
pow|
mult|
genElStr3|
eraseElStr3|
searchElStr3|
eraseSetStr3)
([[segc_1_: <операнд оператора>,]
1_: <операнд оператора>,
[segc_2_: <операнд оператора>,]
2_: <операнд оператора>,
[segc_3_: <операнд оператора>,]
3_: <операнд оператора>][<переходы>])

<оператор вида ElStr5> ::=

(genElStr5|
searchElStr5|
eraseElStr5)
([[segc_1_: <операнд оператора>,]
1_: <операнд оператора>,
[segc_2_: <операнд оператора>,]
2_: <операнд оператора>,
[segc_3_: <операнд оператора>,]
3_: <операнд оператора>,
[segc_4_: <операнд оператора>,]
4_: <операнд оператора>,
[segc_5_: <операнд оператора>,]
5_: <операнд оператора>][<переходы>])

<оператор вида SetStr3> ::=

(searchSetStr3|
selectYStr3|
selectNStr3)
([[[segc_set1_: <операнд оператора>,]
set1_: <операнд оператора>,]
[segc_1_: <операнд оператора>,]
1_: <операнд оператора>,

[[segc_set2_: <операнд оператора>,]
set2_: <операнд оператора>,]
[segc_2_: <операнд оператора>,]
2_: <операнд оператора>,
[[segc_set3_: <операнд оператора>,]
set3_: <операнд оператора>,]
[segc_3_: <операнд оператора>,]
3_: <операнд оператора>][<переходы>])

<оператор вида SetStr5> ::=

(searchSetStr5|
selectYStr5|
selectNStr5)
([[[segc_set1_: <операнд оператора>,]
set1_: <операнд оператора>,]
[segc_1_: <операнд оператора>,]
1_: <операнд оператора>,

[[segc_set2_: <операнд оператора>,]
set2_: <операнд оператора>,]
[segc_2_: <операнд оператора>,]
2_: <операнд оператора>,
[[segc_set3_: <операнд оператора>,]
set3_: <операнд оператора>,]
[segc_3_: <операнд оператора>,]
3_: <операнд оператора>,

[[segc_set4_: <операнд оператора>,]
set4_: <операнд оператора>,]
[segc_4_: <операнд оператора>,]
4_: <операнд оператора>,

[[segc_set5_: <операнд оператора>,]
set5_: <операнд оператора>,]
[segc_5_: <операнд оператора>,]
5_: <операнд оператора>][<переходы>])

<переходы> ::=

,[<идентификатор>][,<идентификатор>]

<операнд оператора> ::=

(assign_| fixed_):[(const_| var_| meta_):][(pos_| fuz_| neg_):][f _:]
(<идентификатор>| [<система множеств>]| [<связка>])

<идентификатор> ::=

(<scs-идентификатор>[1]| [<зарезервированное слово m4scp>])

<зарезервированное слово m4scp> ::=

(__line__| __file__| builtin| end| return|
m4incr| m4ifelse| m4popdef| m4pushdef| m4define| m4undefine|
m4include| m4wrap| m4exit| m4scp_define_operator| M4SCP<идентификатор>)

[1] См. scs.doc
